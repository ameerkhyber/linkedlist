struct node{

	int data;
	struct node* next;

};


struct node* insertatbeg(struct node* head,int x)
{ 
struct node* temp = (struct node*)malloc(sizeof(struct node));
temp->next = NULL;
temp->data = x;

if(head==NULL)
{
	head = temp;
	return head;
}
else
{
	temp->next = head;
	head = temp;
	return head;
}

}
